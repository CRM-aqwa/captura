// ═══════════════════════════════════════════════════════════ // CAPTURA — Récapitulatif hebdomadaire automatique // Google Apps Script — à installer sur script.google.com // ═══════════════════════════════════════════════════════════ // // INSTALLATION (5 minutes) : // 1. Va sur https://script.google.com // 2. Nouveau projet → colle ce code // 3. Remplace EMAIL_DESTINATAIRE par ton email // 4. Clique sur "Exécuter" une fois pour autoriser les accès // 5. Déclencheurs → Ajouter un déclencheur : // Fonction : envoyerRecap // Type : basé sur le temps → hebdomadaire → lundi → 8h-9h // ═══════════════════════════════════════════════════════════ const EMAIL_DESTINATAIRE = 'lack.regis@gmail.com'; // ← ton email const NOM = 'Régis'; function envoyerRecap() { const data = lireDonneesCaptura(); if (!data) { Logger.log('Impossible de lire les données Captura depuis Drive'); return; } const items = data.items || []; const now = new Date(); // Calcul semaine lundi-dimanche const monday = new Date(now); monday.setDate(now.getDate() - ((now.getDay() + 6) % 7)); monday.setHours(0, 0, 0, 0); const sunday = new Date(monday); sunday.setDate(monday.getDate() + 6); sunday.setHours(23, 59, 59, 999); const mondayISO = monday.toISOString().split('T')[0]; const sundayISO = sunday.toISOString().split('T')[0]; const todayISO = now.toISOString().split('T')[0]; // Filtrer les tâches const actives = items.filter(i => !i.done && i.type !== 'idea'); const overdue = actives.filter(i => i.date && i.date < todayISO) .sort((a, b) => a.date.localeCompare(b.date)); const weekTasks = actives.filter(i => i.date && i.date >= mondayISO && i.date <= sundayISO) .sort((a, b) => a.date.localeCompare(b.date) || (a.time||'').localeCompare(b.time||'')); const urgentNoDate = actives.filter(i => i.priority === 'urgent' && !i.date); const allUrgent = actives.filter(i => i.priority === 'urgent'); // Ne pas envoyer si rien à signaler if (overdue.length === 0 && weekTasks.length === 0 && urgentNoDate.length === 0) { Logger.log('Rien à signaler cette semaine — email non envoyé'); return; } const sujet = `Captura — Semaine du ${fmtDate(mondayISO)} (${overdue.length > 0 ? '⚠️ ' + overdue.length + ' en retard · ' : ''}${weekTasks.length} cette semaine)`; const corps = genererHTML(mondayISO, sundayISO, overdue, weekTasks, urgentNoDate, actives.length, items.filter(i => i.type === 'idea' && !i.done).length, now); MailApp.sendEmail({ to: EMAIL_DESTINATAIRE, subject: sujet, htmlBody: corps, name: 'Captura' }); Logger.log('Récapitulatif envoyé à ' + EMAIL_DESTINATAIRE); } // ── LECTURE DES DONNÉES DEPUIS DRIVE APPDATA ────────────── function lireDonneesCaptura() { try { // Chercher le fichier dans Drive AppData de l'utilisateur courant const files = DriveApp.getFilesByName('captura_items.json'); if (!files.hasNext()) { // Chercher dans tous les fichiers accessibles const allFiles = DriveApp.searchFiles("title = 'captura_items.json'"); if (!allFiles.hasNext()) { Logger.log('Fichier captura_items.json introuvable'); return null; } const file = allFiles.next(); const content = file.getBlob().getDataAsString(); return { items: JSON.parse(content) }; } const file = files.next(); const content = file.getBlob().getDataAsString(); return { items: JSON.parse(content) }; } catch(e) { Logger.log('Erreur lecture Drive : ' + e.message); return null; } } // ── FORMATAGE DES DATES ─────────────────────────────────── function fmtDate(iso) { if (!iso) return ''; const [y, m, d] = iso.split('-'); return d + '.' + m + '.' + y; } // ── GÉNÉRATION HTML EMAIL ───────────────────────────────── function genererHTML(mondayISO, sundayISO, overdue, weekTasks, urgentNoDate, totalTasks, totalIdeas, now) { function itemCard(item, accentColor) { const dateStr = item.date ? `📅 ${fmtDate(item.date)}${item.time ? ' à ' + item.time : ''}` : ''; const proj = item.project && item.project !== 'Autre' ? item.project : ''; const tags = (item.tags || []).slice(0, 3); return `
${escHtml(item.title)}
${item.summary ? `
${escHtml(item.summary)}
` : ''}
${dateStr ? `${dateStr}` : ''} ${proj ? `${escHtml(proj)}` : ''} ${tags.map(t => `${escHtml(t)}`).join('')}
`; } function section(title, icon, accentColor, bgColor, items) { if (items.length === 0) return ''; return `
${icon} ${title} ${items.length}
${items.map(i => itemCard(i, accentColor)).join('')}
`; } return `
Captura
Bonjour ${NOM} — semaine du ${fmtDate(mondayISO)} au ${fmtDate(sundayISO)}
${totalTasks}
Tâches
${overdue.length}
En retard
${totalIdeas}
Idées
${section('En retard', '⚠️', '#cc2200', '#fff0ee', overdue)} ${section('Cette semaine', '📅', '#0060cc', '#eef4ff', weekTasks)} ${section('Urgent sans date', '🔴', '#cc2200', '#fff0ee', urgentNoDate)} ${(overdue.length + weekTasks.length + urgentNoDate.length) === 0 ? `
✅
Tout est à jour !
Bonne semaine Régis.
` : ''}
Envoyé automatiquement par Captura · ${now.toLocaleDateString('fr-CH')} à ${now.toLocaleTimeString('fr-CH', {hour:'2-digit', minute:'2-digit'})}
`; } function escHtml(s) { return (s || '').replace(/&/g,'&').replace(//g,'>').replace(/"/g,'"'); } // ── TEST MANUEL ─────────────────────────────────────────── // Pour tester sans attendre lundi, exécute cette fonction : function testerEnvoi() { envoyerRecap(); }
